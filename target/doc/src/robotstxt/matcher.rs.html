23/// Instead of just maintaining a Boolean indicating whether a given line has
24/// matched, we maintain a count of the maximum number of characters matched by
25/// that pattern.
26///
27/// This structure stores the information associated with a match (e.g. when a
28/// Disallow is matched) as priority of the match and line matching.
29///
30/// The priority is initialized with a negative value to make sure that a match
31/// of priority 0 is higher priority than no match at all.
32struct Match {
33    priority: i32,
34    line: u32,
35}
36
37impl Default for Match {
38    fn default() -> Self {
39        Match::new(Self::NO_MATCH_PRIORITY, 0)
40    }
41}
42
43impl Match {
44    const NO_MATCH_PRIORITY: i32 = -1;
45    pub fn new(priority: i32, line: u32) -> Match {
46        Match { priority, line }
47    }
48
49    pub fn set(&mut self, priority: i32, line: u32) {
50        self.priority = priority;
51        self.line = line;
52    }
53
54    pub fn clear(&mut self) {
55        self.set(Self::NO_MATCH_PRIORITY, 0);
56    }
57
58    pub fn line(&self) -> u32 {
59        self.line
60    }
61
62    pub fn priority(&self) -> i32 {
63        self.priority
64    }
65
66    pub fn higher_priority_match<'a>(a: &'a Match, b: &'a Match) -> &'a Match {
67        if a.priority() > b.priority() {
68            a
69        } else {
70            b
71        }
72    }
73}

88/// Create a RobotsMatcher with the default matching strategy.
89///
90/// The default matching strategy is longest-match as opposed to the former internet draft
91/// that provisioned first-match strategy. Analysis shows that longest-match,
92/// while more restrictive for crawlers, is what webmasters assume when writing
93/// directives. For example, in case of conflicting matches (both Allow and
94/// Disallow), the longest match is the one the user wants. For example, in
95/// case of a robots.txt file that has the following rules
96/// ```txt
97///   Allow: /
98///   Disallow: /cgi-bin
99/// ```
100/// it's pretty obvious what the webmaster wants: they want to allow crawl of
101/// every URI except /cgi-bin. However, according to the expired internet
102/// standard, crawlers should be allowed to crawl everything with such a rule.
103pub trait RobotsMatchStrategy {
104    fn match_allow(&self, path: &str, pattern: &str) -> i32;
105
106    fn match_disallow(&self, path: &str, pattern: &str) -> i32;
107
108    /// Returns true if URI path matches the specified pattern. Pattern is anchored
109    /// at the beginning of path. '$' is special only at the end of pattern.
110    ///
111    /// Since 'path' and 'pattern' are both externally determined (by the webmaster),
112    /// we make sure to have acceptable worst-case performance.
113    /// ```rust
114    /// use robotstxt::matcher::{LongestMatchRobotsMatchStrategy, RobotsMatchStrategy};
115    ///
116    /// type Target = LongestMatchRobotsMatchStrategy;
117    /// assert_eq!(true, Target::matches("/", "/"));
118    /// assert_eq!(true, Target::matches("/abc", "/"));
119    /// assert_eq!(false, Target::matches("/", "/abc"));
120    /// assert_eq!(
121    ///     true,
122    ///     Target::matches("/google/robotstxt/tree/master", "/*/*/tree/master")
123    /// );
124    /// assert_eq!(
125    ///     true,
126    ///     Target::matches(
127    ///         "/google/robotstxt/tree/master/index.html",
128    ///         "/*/*/tree/master",
129    ///     )
130    /// );
131    /// assert_eq!(
132    ///     true,
133    ///     Target::matches("/google/robotstxt/tree/master", "/*/*/tree/master$")
134    /// );
135    /// assert_eq!(
136    ///     false,
137    ///     Target::matches("/google/robotstxt/tree/master/abc", "/*/*/tree/master$")
138    /// );
139    /// assert_eq!(
140    ///     false,
141    ///     Target::matches("/google/robotstxt/tree/abc", "/*/*/tree/master")
142    /// );
143    /// ```
144    fn matches(path: &str, pattern: &str) -> bool {
145        let pathlen = path.len();
146        let mut pos = Vec::with_capacity(pathlen + 1);
147
148        // The pos[] array holds a sorted list of indexes of 'path', with length
149        // 'numpos'.  At the start and end of each iteration of the main loop below,
150        // the pos[] array will hold a list of the prefixes of the 'path' which can
151        // match the current prefix of 'pattern'. If this list is ever empty,
152        // return false. If we reach the end of 'pattern' with at least one element
153        // in pos[], return true.
154        let mut numpos: usize = 1;
155        pos.insert(0, 0);
156
157        for (index, pat) in pattern.chars().enumerate() {
158            if pat == '$' && index + 1 == pattern.len() {
159                return pos[numpos - 1] == pathlen;
160            }
161
162            if pat == '*' {
163                numpos = pathlen - pos[0] + 1;
164                for i in 1..numpos {
165                    pos.insert(i, pos[i - 1] + 1);
166                }
167            } else {
168                // Includes '$' when not at end of pattern.
169                let mut new_numpos = 0;
170                for i in 0..numpos {
171                    // TODO Optimize chars().nth() ?
172                    if pos[i] < pathlen && path.chars().nth(pos[i]) == Some(pat) {
173                        pos.insert(new_numpos, pos[i] + 1);
174                        new_numpos += 1;
175                    }
176                }
177                numpos = new_numpos;
178
179                if numpos == 0 {
180                    return false;
181                }
182            }
183        }
184        true
185    }
186}

188/// Implements the default robots.txt matching strategy. The maximum number of
189/// characters matched by a pattern is returned as its match priority.
190#[derive(Default)]
191pub struct LongestMatchRobotsMatchStrategy;
192
193impl RobotsMatchStrategy for LongestMatchRobotsMatchStrategy {
194    fn match_allow(&self, path: &str, pattern: &str) -> i32 {
195        if Self::matches(path, pattern) {
196            pattern.len() as i32
197        } else {
198            -1
199        }
200    }
201
202    fn match_disallow(&self, path: &str, pattern: &str) -> i32 {
203        if Self::matches(path, pattern) {
204            pattern.len() as i32
205        } else {
206            -1
207        }
208    }
209}

211/// RobotsMatcher - matches robots.txt against URLs.
212///
213/// The Matcher uses a default match strategy for Allow/Disallow patterns which
214/// is the official way of Google crawler to match robots.txt. It is also
215/// possible to provide a custom match strategy.
216///
217/// The entry point for the user is to call one of the [allowed_by_robots](RobotsMatcher::allowed_by_robots())
218/// methods that return directly if a URL is being allowed according to the
219/// robots.txt and the crawl agent.
220/// The RobotsMatcher can be re-used for URLs/robots.txt but is not thread-safe.
221#[derive(Default)]
222pub struct RobotsMatcher<'a, S: RobotsMatchStrategy> {
223    /// Characters of 'url' matching Allow.
224    allow: MatchHierarchy,
225    /// Characters of 'url' matching Disallow.
226    disallow: MatchHierarchy,
227    /// True if processing global agent rules.
228    seen_global_agent: bool,
229    /// True if processing our specific agent.
230    seen_specific_agent: bool,
231    /// True if we ever saw a block for our agent.
232    ever_seen_specific_agent: bool,
233    /// True if saw any key: value pair.
234    seen_separator: bool,
235    /// The path we want to pattern match. Not owned and only a valid pointer
236    /// during the lifetime of [allowed_by_robots](RobotsMatcher::allowed_by_robots()) calls.
237    path: Cow<'a, str>,
238    /// The User-Agents we are interested in. Not owned and only a valid
239    /// pointer during the lifetime of [allowed_by_robots](RobotsMatcher::allowed_by_robots()) calls.
240    user_agents: Vec<&'a str>,
241    match_strategy: S,
242}
243
244impl<'a, S: RobotsMatchStrategy> RobotsMatcher<'a, S> {
245    /// Initialize next path and user-agents to check. Path must contain only the
246    /// path, params, and query (if any) of the url and must start with a '/'.
247    fn init_user_agents_and_path(&mut self, user_agents: Vec<&'a str>, path: Cow<'a, str>) {
248        self.path = path;
249        self.user_agents = user_agents;
250    }
251
252    /// Returns true if 'url' is allowed to be fetched by any member of the
253    /// "user_agents" vector. 'url' must be %-encoded according to RFC3986.
254    pub fn allowed_by_robots(
255        &mut self,
256        robots_body: &'a str,
257        user_agents: Vec<&'a str>,
258        url: &'a str,
259    ) -> bool
260    where
261        Self: RobotsParseHandler,
262    {
263        // The url is not normalized (escaped, percent encoded) here because the user
264        // is asked to provide it in escaped form already.
265        let path = super::get_path_params_query(url);
266        self.init_user_agents_and_path(user_agents, path);
267        super::parse_robotstxt(robots_body, self);
268        !self.disallow()
269    }
270
271    /// Do robots check for 'url' when there is only one user agent. 'url' must
272    /// be %-encoded according to RFC3986.
273    pub fn one_agent_allowed_by_robots(
274        &mut self,
275        robots_txt: &'a str,
276        user_agent: &'a str,
277        url: &'a str,
278    ) -> bool
279    where
280        Self: RobotsParseHandler,
281    {
282        self.allowed_by_robots(robots_txt, vec![user_agent], url)
283    }
284
285    /// Returns true if we are disallowed from crawling a matching URI.
286    fn disallow(&self) -> bool {
287        if self.allow.specific.priority() > 0 || self.disallow.specific.priority() > 0 {
288            return self.disallow.specific.priority() > self.allow.specific.priority();
289        }
290
291        if self.ever_seen_specific_agent {
292            // Matching group for user-agent but either without disallow or empty one,
293            // i.e. priority == 0.
294            return false;
295        }
296
297        if self.disallow.global.priority() > 0 || self.allow.global.priority() > 0 {
298            return self.disallow.global.priority() > self.allow.global.priority();
299        }
300
301        false
302    }
303
304    /// Returns true if any user-agent was seen.
305    fn seen_any_agent(&self) -> bool {
306        self.seen_global_agent || self.seen_specific_agent
307    }
308
309    /// Extract the matchable part of a user agent string, essentially stopping at
310    /// the first invalid character.
311    /// Example: 'Googlebot/2.1' becomes 'Googlebot'
312    fn extract_user_agent(user_agent: &str) -> &str {
313        // Allowed characters in user-agent are [a-zA-Z_-].
314        if let Some(end) =
315            user_agent.find(|c: char| !(c.is_ascii_alphabetic() || c == '-' || c == '_'))
316        {
317            &user_agent[..end]
318        } else {
319            user_agent
320        }
321    }
322
323    /// Verifies that the given user agent is valid to be matched against
324    /// robots.txt. Valid user agent strings only contain the characters
325    /// [a-zA-Z_-].
326    pub fn is_valid_user_agent_to_obey(user_agent: &str) -> bool {
327        !user_agent.is_empty() && Self::extract_user_agent(user_agent) == user_agent
328    }
329
330    /// Returns true if we are disallowed from crawling a matching URI. Ignores any
331    /// rules specified for the default user agent, and bases its results only on
332    /// the specified user agents.
333    fn disallow_ignore_global(&self) -> bool {
334        if self.allow.specific.priority() > 0 || self.disallow.specific.priority() > 0 {
335            return self.disallow.specific.priority() > self.allow.specific.priority();
336        }
337        false
338    }
339
340    /// Returns the line that matched or 0 if none matched.
341    fn matching_line(&self) -> u32 {
342        if self.ever_seen_specific_agent {
343            return Match::higher_priority_match(&self.disallow.specific, &self.allow.specific)
344                .line();
345        }
346        Match::higher_priority_match(&self.disallow.global, &self.allow.global).line()
347    }
348}

350impl<S: RobotsMatchStrategy> RobotsParseHandler for RobotsMatcher<'_, S> {
351    fn handle_robots_start(&mut self) {
352        // This is a new robots.txt file, so we need to reset all the instance member
353        // variables. We do it in the same order the instance member variables are
354        // declared, so it's easier to keep track of which ones we have (or maybe
355        // haven't!) done.
356        self.allow.clear();
357        self.disallow.clear();
358
359        self.seen_global_agent = false;
360        self.seen_specific_agent = false;
361        self.ever_seen_specific_agent = false;
362        self.seen_separator = false;
363    }
364
365    fn handle_robots_end(&mut self) {}
366
367    fn handle_user_agent(&mut self, line_num: u32, user_agent: &str) {
368        if self.seen_separator {
369            self.seen_specific_agent = false;
370            self.seen_global_agent = false;
371            self.seen_separator = false;
372        }
373
374        // Google-specific optimization: a '*' followed by space and more characters
375        // in a user-agent record is still regarded a global rule.
376        if !user_agent.is_empty()
377            && user_agent.starts_with('*')
378            && (user_agent.len() == 1 || user_agent[1..].starts_with(char::is_whitespace))
379        {
380            self.seen_global_agent = true;
381        } else {
382            let user_agent = Self::extract_user_agent(user_agent);
383            for agent in &self.user_agents {
384                if user_agent.eq_ignore_ascii_case(&agent) {
385                    self.ever_seen_specific_agent = true;
386                    self.seen_specific_agent = true;
387                    break;
388                }
389            }
390        }
391    }
392
393    fn handle_allow(&mut self, line_num: u32, value: &str) {
394        if !self.seen_any_agent() {
395            return;
396        }
397
398        self.seen_separator = true;
399        let priority = self.match_strategy.match_disallow(&self.path, value);
400        if priority >= 0 {
401            if self.seen_specific_agent {
402                if self.allow.specific.priority() < priority {
403                    self.allow.specific.set(priority, line_num);
404                }
405            } else if self.allow.global.priority() < priority {
406                self.allow.global.set(priority, line_num);
407            }
408        } else {
409            // Google-specific optimization: 'index.htm' and 'index.html' are normalized to '/'.
410            let slash_pos = value.rfind('/');
411
412            if let Some(slash_pos) = slash_pos {
413                if value[slash_pos..].starts_with("/index.htm") {
414                    let new_pattern = format!("{}{}", &value[..(slash_pos + 1)], "$");
415                    self.handle_allow(line_num, &new_pattern);
416                }
417            }
418        }
419    }
420
421    fn handle_disallow(&mut self, line_num: u32, value: &str) {
422        if !self.seen_any_agent() {
423            return;
424        }
425
426        self.seen_separator = true;
427        let priority = self.match_strategy.match_disallow(&self.path, value);
428        if priority >= 0 {
429            if self.seen_specific_agent {
430                if self.disallow.specific.priority() < priority {
431                    self.disallow.specific.set(priority, line_num);
432                }
433            } else if self.disallow.global.priority() < priority {
434                self.disallow.global.set(priority, line_num);
435            }
436        }
437    }
438
439    fn handle_sitemap(&mut self, line_num: u32, value: &str) {
440        self.seen_separator = true;
441    }
442
443    fn handle_unknown_action(&mut self, line_num: u32, action: &str, value: &str) {
444        self.seen_separator = true;
445    }
446}

448#[cfg(test)]
449mod test {
450    use crate::matcher::*;
451
452    #[test]
453    fn test_extract_user_agent<'a>() {
454        // Example: 'Googlebot/2.1' becomes 'Googlebot'
455        type Target<'a> = RobotsMatcher<'a, LongestMatchRobotsMatchStrategy>;
456        assert_eq!("Googlebot", Target::extract_user_agent("Googlebot/2.1"));
457        assert_eq!("Googlebot", Target::extract_user_agent("Googlebot"));
458        assert_eq!("Googlebot-", Target::extract_user_agent("Googlebot-"));
459        assert_eq!("Googlebot_", Target::extract_user_agent("Googlebot_"));
460        assert_eq!("Googlebot_", Target::extract_user_agent("Googlebot_2.1"));
461        assert_eq!("", Target::extract_user_agent("1Googlebot_2.1"));
462        assert_eq!("Goo", Target::extract_user_agent("Goo1glebot_2.1"));
463    }
